parseHeader :: Parser [ByteString]

Parses the header of a FASTQ file.

The header starts with a @ character and follows with an optional description. When making use if BioSeq, the first word of the header is used as the seqid.

parseFastQ :: String -> Result [FastQSequence]

Parses sequences from a String.

parseFastQ x = parseString parseSequences mempty x

parseFastQB :: ByteString -> Result [FastQSequence]

Parses sequences from a strict ByteString.

parseFastQB x = parseByteString parseSequences mempty x
